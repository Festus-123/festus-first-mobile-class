import React, { useState } from "react";
import {
  View,
  Text,
  TextInput,
  TouchableOpacity,
  StyleSheet,
  SafeAreaView,
  // Vibration,
} from "react-native";
import { LinearGradient } from "expo-linear-gradient";
import { Ionicons } from "@expo/vector-icons";
// import AsyncStorage from "@react-native-async-storage

export default function SignUpScreen({ navigation }) {
  const [username, setUsername] = useState("");
  const [password, setPassword] = useState("");
  const [confirmPassword, setConfirmPassword] = useState("");

  return (
    <SafeAreaView style={styles.container}>
      <Text style={styles.title}>Create an account</Text>

      <View style={styles.inputContainer}>
        <Ionicons
          name="person-outline"
          size={20}
          color="gray"
          style={styles.icon}
        />
        <TextInput
          style={styles.input}
          placeholder="Username"
          value={username}
          onChangeText={setUsername}
        />
      </View>
      <Text style={styles.hint}>A minimum of 4 characters</Text>


      <View style={styles.inputContainer}>
        <Ionicons
          name="lock-closed-outline"
          size={20}
          color="gray"
          style={styles.icon}
        />
        <TextInput
          style={styles.input}
          placeholder="Password"
          secureTextEntry
          value={password}
          onChangeText={setPassword}
        />
      </View>
      <Text style={styles.hint}>A minimum of 8 characters</Text>
      <Text style={styles.hint}>One uppercase letter</Text>


      <View style={styles.inputContainer}>
        <Ionicons
          name="lock-closed"
          size={20}
          color="gray"
          style={styles.icon}
        />
        <TextInput
          style={styles.input}
          placeholder="Confirm Password"
          secureTextEntry
          value={confirmPassword}
          onChangeText={setConfirmPassword}
        />
      </View>
      <Text style={styles.hint}>
        Your confirm password should be the same as password
      </Text>


      <TouchableOpacity style={{ marginTop: 20 }} >
        <LinearGradient colors={["#ff7e5f", "#feb47b"]} style={styles.button}>
          <Text style={styles.buttonText}>Sign Up</Text>
        </LinearGradient>
      </TouchableOpacity>


      <View style={styles.footer}>
        <Text style={styles.footerText}>Already have an account? </Text>
        <TouchableOpacity onPress={() => navigation.navigate("SignIn")}>
          <Text style={styles.signIn}>Sign in</Text>
        </TouchableOpacity>
      </View>
    </SafeAreaView>
  );
}

const styles = StyleSheet.create({
  container:
   { flex: 1, backgroundColor: "#fff", padding: 20, marginTop: 50, gap: 25 },
  title: 
  { fontSize: 24, fontWeight: "bold", marginBottom: 20, textAlign: "center" },
  inputContainer:
   {
    flexDirection: "row", alignItems: "center", borderBottomWidth: 2, borderColor: "#ccc",
    marginTop: 10, paddingVertical: 5,
  },
  icon:
   { marginRight: 10 },
  input:
   { flex: 1, paddingVertical: 8, fontSize: 16 },
  button:
   { paddingVertical: 15, borderRadius: 8, alignItems: "center" },
  buttonText:
   { color: "#fff", fontSize: 18, fontWeight: "bold" },
  footer: 
  { flexDirection: "row", justifyContent: "center", marginTop: 30 },
  footerText:
   { fontSize: 14, color: "#555" },
  signIn:
   { fontSize: 14, color: "#ff7e5f", fontWeight: "bold" },
  hint: 
  { fontSize: 12, color: "gray", marginLeft: 5, marginTop: 3 },
});
